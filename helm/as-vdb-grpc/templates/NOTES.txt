{{- if .Values.ftlserver.enabled }}
   FTL Server:
{{- if eq .Values.ftlserver.service.type "NodePort" }}
   export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "as-vdb-grpc.fullname" . }}-ftlserver)
   export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
   echo "FTL Server available at $NODE_IP:$NODE_PORT"
{{- else if eq .Values.ftlserver.service.type "LoadBalancer" }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "as-vdb-grpc.fullname" . }}-ftlserver'
   export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "as-vdb-grpc.fullname" . }}-ftlserver --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
   echo "FTL Server available at $SERVICE_IP:{{ .Values.ftlserver.service.port }}"
{{- else if eq .Values.ftlserver.service.type "ClusterIP" }}
   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "as-vdb-grpc.fullname" . }}-ftlserver {{ .Values.ftlserver.service.port }}:{{ .Values.ftlserver.service.port }}
   echo "FTL Server available at 127.0.0.1:{{ .Values.ftlserver.service.port }}"
{{- end }}
{{- end }}

2. Check the status of your deployment:
   kubectl get pods --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}

3. View logs:
   # All services
   kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}
   
   # Specific services
   kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/component=ftlserver
   kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/component=vectorstore  
   kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/component=documentstore

4. To scale the TIBDG components:
   kubectl scale deployment {{ include "as-vdb-grpc.fullname" . }}-tibdgproxy --namespace {{ .Release.Namespace }} --replicas=2
   kubectl scale deployment {{ include "as-vdb-grpc.fullname" . }}-tibdgnode --namespace {{ .Release.Namespace }} --replicas=3
